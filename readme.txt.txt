1.konfiguracja znajduje się w pliku config.php
	$db_host = host bazy danych
	$db_name = nazwa bazy danych
	$db_user = użytkownik bazy danych
	$db_password = hasło bazy danych
	$losowana_liczba_fiszek = ilość wyświetlanych fiszek, domyślnie 12
	$poprawna_odpowiedź = ilość punktów za poprawną odpowiedź, domyślnie 2
	$niepoprawna_odpowiedź = ilość ponktów za niepoprawną odpowiedź domyślnie -1

2.baza.sql zawiea dump bazy danych
